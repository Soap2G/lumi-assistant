## Context

The skill library at `config/` currently contains 14 skills + 2 agents
(VERSION = 1.9.0). The eval infrastructure has two scripts:

  names, eval references). Runs in CI, no network.

Write this file **verbatim**:

```python
#!/usr/bin/env python3
"""Content-level health check for the skill library.

Complements lint.py (structural validation) and run.py (routing
accuracy) with checks that detect drift between skills, AGENTS.md,
and eval cases.

Offline checks (no network, safe for CI):
  1. Cross-reference integrity — every skill/agent name in a
     NOT-covered redirect (``use `name```) exists in the library.
  2. AGENTS.md skill references — every ``load the `name` skill``
     directive in guidelines points to an existing skill.
  3. Eval coverage — every registered skill/agent appears in at
     least one should_match entry in cases.yaml.

Online checks (require network, use --online):
  4. URL liveness — HEAD-request https:// URLs in SKILL.md files.

Run from the repo root:

    python config/evals/health.py            # offline only
    python config/evals/health.py --online   # include URL probes
"""
from __future__ import annotations

import re
import sys
from pathlib import Path

import yaml

ROOT = Path(__file__).resolve().parents[2]
SKILLS_DIR = ROOT / "config" / "skills"
AGENTS_DIR = ROOT / "config" / "agents"
AGENTS_MD = ROOT / "config" / "AGENTS.md"
CASES_FILE = ROOT / "config" / "evals" / "cases.yaml"


def collect_names() -> dict[str, Path]:
    """Return {name: path} for all registered skills and agents."""
    names: dict[str, Path] = {}
    for path in SKILLS_DIR.rglob("SKILL.md"):
        text = path.read_text(encoding="utf-8")
        m = re.search(r"^name:\s*(.+?)\s*$", text, flags=re.MULTILINE)
        if m:
            names[m.group(1).strip()] = path
    for path in AGENTS_DIR.glob("*.md"):
        names[path.stem] = path
    return names


def check_crossrefs(names: dict[str, Path]) -> list[str]:
    """Verify that ``use `name``` redirects in skill files point to real entries."""
    errors: list[str] = []
    # "use `name`", "use the `name`", "or `name`" (in NOT-covered lists)
    pat = re.compile(
        r"(?:use|see)\s+(?:the\s+)?`([a-z][-a-z0-9]*)`"
        r"|"
        r"or\s+`([a-z][-a-z0-9]*)`"
    )
    for name, path in sorted(names.items()):
        if not str(path).endswith("SKILL.md"):
            continue
        text = path.read_text(encoding="utf-8")
        rel = path.relative_to(ROOT)
        for m in pat.finditer(text):
            ref = m.group(1) or m.group(2)
            if ref and ref not in names:
                errors.append(f"{rel}: references unknown name `{ref}`")
    return errors


def check_agents_md_refs(names: dict[str, Path]) -> list[str]:
    """Verify that AGENTS.md skill-loading directives point to real entries."""
    errors: list[str] = []
    text = AGENTS_MD.read_text(encoding="utf-8")
    patterns = [
        re.compile(r"load\s+(?:the\s+)?`([a-z][-a-z0-9]*)`"),
        re.compile(r"the\s+`([a-z][-a-z0-9]*)`\s+skill"),
    ]
    seen = set()
    for pat in patterns:
        for m in pat.finditer(text):
            ref = m.group(1)
            if ref not in seen and ref not in names:
                errors.append(f"AGENTS.md: references unknown skill `{ref}`")
                seen.add(ref)
    return errors


def check_eval_coverage(names: dict[str, Path]) -> list[str]:
    """Every registered skill/agent should have at least one should_match entry."""
    errors: list[str] = []
    cases = yaml.safe_load(CASES_FILE.read_text(encoding="utf-8")) or {}
    covered = set()
    for case in cases.get("should_match", []):
        truth = case.get("skill") or case.get("agent")
        if truth:
            covered.add(truth)
    for name in sorted(names):
        if name not in covered:
            errors.append(f"cases.yaml: no should_match entry for `{name}`")
    return errors


def check_urls(names: dict[str, Path]) -> list[str]:
    """HEAD-request https:// URLs in SKILL.md files."""
    import httpx

    errors: list[str] = []
    url_re = re.compile(r"https://[^\s)>`|]+")
    urls: dict[str, list[str]] = {}
    for name, path in sorted(names.items()):
        if not str(path).endswith("SKILL.md"):
            continue
        text = path.read_text(encoding="utf-8")
        rel = str(path.relative_to(ROOT))
        for m in url_re.finditer(text):
            url = m.group(0).rstrip(".,;:")
            if "<" in url and ">" in url:
                continue
            urls.setdefault(url, []).append(rel)

    print(f"  Probing {len(urls)} unique URLs...")
    client = httpx.Client(follow_redirects=True, timeout=15)
    for url, sources in sorted(urls.items()):
        try:
            resp = client.head(url)
            if resp.status_code >= 400:
                errors.append(
                    f"URL {resp.status_code}: {url} (in {', '.join(sources)})"
                )
        except httpx.HTTPError as exc:
            errors.append(f"URL unreachable: {url} ({exc.__class__.__name__})")
    client.close()
    return errors


def main() -> int:
    online = "--online" in sys.argv
    names = collect_names()
    print(f"Health check — {len(names)} entries, online={'yes' if online else 'no'}")

    errors: list[str] = []

    print("\n1. Cross-reference integrity...")
    errs = check_crossrefs(names)
    errors.extend(errs)
    print(f"   {'PASS' if not errs else f'{len(errs)} error(s)'}")

    print("2. AGENTS.md skill references...")
    errs = check_agents_md_refs(names)
    errors.extend(errs)
    print(f"   {'PASS' if not errs else f'{len(errs)} error(s)'}")

    print("3. Eval coverage...")
    errs = check_eval_coverage(names)
    errors.extend(errs)
    print(f"   {'PASS' if not errs else f'{len(errs)} error(s)'}")

    if online:
        print("4. URL liveness...")
        errs = check_urls(names)
        errors.extend(errs)
        print(f"   {'PASS' if not errs else f'{len(errs)} error(s)'}")

    if errors:
        print(f"\n{len(errors)} error(s):")
        for e in errors:
            print(f"  - {e}")
        return 1
    print("\nAll health checks pass.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

## Principle 9: upstream-change propagation

Skills contain curated knowledge derived from upstream sources: PDG
editions, INSPIRE/HEPData APIs, ATLAS Open Data releases, REANA/Rucio

## v1.12.0 — Auto-generated library index

### 1. New file: `config/evals/index.py`

Write this file **verbatim**:

```python
#!/usr/bin/env python3
"""Generate the library index (config/INDEX.md).

Walks config/skills/ and config/agents/, extracts name, scope, and a
one-line summary from each, and writes a markdown overview organized
by intent category. Run after adding, removing, or renaming a skill:

    python config/evals/index.py

The output is committed — CI checks that running this script produces
no diff, so the index stays in sync with the library.
"""
from __future__ import annotations

import re
from pathlib import Path

ROOT = Path(__file__).resolve().parents[2]
SKILLS_DIR = ROOT / "config" / "skills"
AGENTS_DIR = ROOT / "config" / "agents"
INDEX_FILE = ROOT / "config" / "INDEX.md"


def extract_frontmatter(text: str) -> str | None:
    m = re.match(r"^---\r?\n(.*?)\r?\n---", text, flags=re.DOTALL)
    return m.group(1) if m else None


def extract_field(fm: str, key: str) -> str | None:
    m = re.search(rf"^{re.escape(key)}:\s*(.+?)\s*$", fm, flags=re.MULTILINE)
    return m.group(1).strip() if m else None


def first_sentence(desc: str) -> str:
    """Extract a short summary from the description."""
    # Strip the "Use when" lead — too repetitive in a table
    cleaned = re.sub(r"^Use when\s+", "", desc)
    # Take up to the first period followed by whitespace
    idx = re.search(r"\.\s", cleaned)
    if idx and idx.start() < 140:
        return cleaned[: idx.start() + 1]
    # Fallback: truncate
    if len(cleaned) > 120:
        return cleaned[:117] + "..."
    return cleaned


def intent_category(path: Path) -> str:
    """Derive the intent category from the path."""
    rel = path.relative_to(SKILLS_DIR)
    parts = rel.parts
    return parts[0] if len(parts) > 2 else "(top-level)"


def main() -> None:
    skills_by_cat: dict[str, list[dict]] = {}
    agents: list[dict] = []

    for path in sorted(SKILLS_DIR.rglob("SKILL.md")):
        text = path.read_text(encoding="utf-8")
        fm = extract_frontmatter(text)
        if not fm:
            continue
        name = extract_field(fm, "name")
        desc = extract_field(fm, "description")
        scope = extract_field(fm, "data_scope") or "?"
        if not name or not desc:
            continue
        cat = intent_category(path)
        skills_by_cat.setdefault(cat, []).append(
            {"name": name, "scope": scope, "short": first_sentence(desc)}
        )

    for path in sorted(AGENTS_DIR.glob("*.md")):
        text = path.read_text(encoding="utf-8")
        fm = extract_frontmatter(text)
        if not fm:
            continue
        desc = extract_field(fm, "description")
        scope_raw = extract_field(fm, "accepts_data_scope") or "?"
        if not desc:
            continue
        agents.append(
            {"name": path.stem, "scope": scope_raw, "short": first_sentence(desc)}
        )

    n_skills = sum(len(v) for v in skills_by_cat.values())
    total = n_skills + len(agents)
    lines = [
        "<!-- Auto-generated by config/evals/index.py — do not edit by hand. -->",
        "",
        "# Skill Library Index",
        "",
        (
            f"{total} entries: {n_skills} skills across"
            f" {len(skills_by_cat)} intent"
            f" {'category' if len(skills_by_cat) == 1 else 'categories'}"
            f", {len(agents)} agent{'s' if len(agents) != 1 else ''}."
        ),
        "",
    ]

    for cat in sorted(skills_by_cat):
        entries = skills_by_cat[cat]
        lines.append(
            f"## {cat}/ ({len(entries)}"
            f" skill{'s' if len(entries) != 1 else ''})"
        )
        lines.append("")
        lines.append("| Skill | Scope | Summary |")
        lines.append("|---|---|---|")
        for e in sorted(entries, key=lambda x: x["name"]):
            lines.append(f"| `{e['name']}` | {e['scope']} | {e['short']} |")
        lines.append("")

    if agents:
        lines.append(f"## Agents ({len(agents)})")
        lines.append("")
        lines.append("| Agent | Scope | Summary |")
        lines.append("|---|---|---|")
        for a in sorted(agents, key=lambda x: x["name"]):
            lines.append(f"| `{a['name']}` | {a['scope']} | {a['short']} |")
        lines.append("")

    INDEX_FILE.write_text("\n".join(lines), encoding="utf-8")
    print(f"Wrote {INDEX_FILE.relative_to(ROOT)} — {total} entries.")


if __name__ == "__main__":
    main()
```
